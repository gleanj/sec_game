    start: {
        description: 'Start training module',
        execute: async (args) => {
            // Handle "start module 1" format
            if (args.length < 2) {
                return `<div class="error-line">Usage: start module [number]</div>
<div class="result-line">Available modules: 1-3</div>
<div class="info-line mt-1">Example: start module 1</div>`;
            }
            
            const keyword = args[0].toLowerCase();
            const moduleNum = args[1];
            
            if (keyword !== 'module') {
                return `<div class="error-line">Usage: start module [number]</div>
<div class="result-line">Available modules: 1-3</div>`;
            }
            
            if (moduleNum === '1') {
                await executeModule1();
                return null;
            } else if (moduleNum === '2') {
                await executeModule2();
                return null;
            } else if (moduleNum === '3') {
                await executeModule3();
                return null;
            } else {
                return `<div class="error-line">Module ${moduleNum} not found</div>
<div class="result-line">Available modules: 1-3</div>`;
            }
        }
    },
